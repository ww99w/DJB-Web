SELECT
LBITNO
,HMITNA
,HMITG1 
,HMITG2 
,HMITG3 
,HMITG4 
,HMITG5 
,SDITST
,CBKYNA
,LBSIZE
,LBWHNO
,HWWHNA
,LBLONO
,HLLONA
,LBYEAR
,LBMONT
,LBEDQT
FROM
(
SELECT 
LBITNO
,HMITNA
, ISNULL(HG1.HGIGNA,'') AS HMITG1 
, ISNULL(HG2.HGIGNA,'') AS HMITG2 
, ISNULL(HG3.HGIGNA,'') AS HMITG3 
, ISNULL(HG4.HGIGNA,'') AS HMITG4 
, ISNULL(HG5.HGIGNA,'') AS HMITG5 
,(SELECT TOP 1
	SDITST
	FROM SSP3
	LEFT JOIN SSP2 ON 1=1
	AND SECONO=SFCONO
	AND SESPDN=SFSPDN
	LEFT JOIN SSP1 ON 1=1
	AND SDCONO=SFCONO
	AND SDSPDN=SFSPDN
	WHERE 1=1 
	AND SFCONO=LBCONO
	AND SEBRNO=LBWHNO
	AND SFITNO=LBITNO
	ORDER BY SDEFDT DESC
	) AS SDITST
,CBKYNA AS LBSIZE
,LBWHNO
,HWWHNA
,LBLONO
,HLLONA
,LBYEAR
,LBMONT
,LBEDQT
FROM IBL2
LEFT JOIN IIMA ON 1=1
AND HMCONO=LBCONO
AND HMBRNO=''
AND HMITNO=LBITNO
LEFT JOIN IGRP HG1 ON 1=1 
AND HG1.HGCONO=HMCONO 
AND UPPER(ISNULL(HG1.HGBRNO,' '))='' 
AND HG1.HGITCA=HMITCA 
AND HG1.HGIGNO=HMITG1 
LEFT JOIN IGRP HG2 ON 1=1 
AND HG2.HGCONO=HMCONO 
AND UPPER(ISNULL(HG2.HGBRNO,' '))='' 
AND HG2.HGITCA=HMITCA 
AND HG2.HGIGNO=HMITG2 
LEFT JOIN IGRP HG3 ON 1=1 
AND HG3.HGCONO=HMCONO 
AND UPPER(ISNULL(HG3.HGBRNO,' '))='' 
AND HG3.HGITCA=HMITCA 
AND HG3.HGIGNO=HMITG3 
LEFT JOIN IGRP HG4 ON 1=1 
AND HG4.HGCONO=HMCONO 
AND UPPER(ISNULL(HG4.HGBRNO,' '))='' 
AND HG4.HGITCA=HMITCA 
AND HG4.HGIGNO=HMITG4 
LEFT JOIN IGRP HG5 ON 1=1 
AND HG5.HGCONO=HMCONO 
AND UPPER(ISNULL(HG5.HGBRNO,' '))='' 
AND HG5.HGITCA=HMITCA 
AND HG5.HGIGNO=HMITG5 
LEFT JOIN IWHS ON 1=1 
AND HWCONO=HMCONO 
AND UPPER(ISNULL(HWBRNO,' '))=' '
AND HWWHNO=LBWHNO
LEFT JOIN ILOC ON 1=1 
AND HLCONO=HMCONO 
AND UPPER(ISNULL(HLBRNO,' '))='' 
AND HLWHNO=LBWHNO 
AND HLLONO=LBLONO 
LEFT JOIN GCT2 AS GC2 ON 1=1 
AND CBRCST=1 
AND UPPER(ISNULL(CBTBNO,' '))='SIZE' 
AND CBKYNO = LBSIZE 
WHERE 1=1
AND LBCONO='DJB'
AND LBBRNO=''
AND LBYEAR=DATEPART(YEAR,GETDATE()) 
AND LBMONT=DATEPART(MONTH,GETDATE()) 
AND LBWHNO='C046'
AND LBLONO='A.01'
AND HMITCA='APL'
) VW1
LEFT JOIN GCT2 ON 1=1
AND CBCONO='DJB'
AND CBBRNO=''
AND CBTBNO='ITST'
AND CBKYNO=SDITST
WHERE SDITST='IS001'
